47  /**
48   * Corresponding to each method, an executor is launched to execute {@link BeforeFirstRun},
49   * {@link BeforeEachRun}, {@link AfterEachRun} and {@link AfterLastRun} classes. To store the data if the
50   * single-execute before
51   * classes have been executed, this class is implemented as a singleton to store
52   * this information related to the method. All the data comes from the {@link BenchmarkMethod} class.
53   * 
54   * @author Sebastian Graf, University of Konstanz
55   */
56  public final class BenchmarkExecutor {
57  
58      /**
59       * Static mapping for all methods to be executed because of the single-runs
60       * before/after methods.
61       */
62      private static final Map<BenchmarkMethod, BenchmarkExecutor> EXECUTOR =
63          new Hashtable<BenchmarkMethod, BenchmarkExecutor>();
64  
65      /** Set with all meters to be benched automatically. */
66      private static final Set<AbstractMeter> METERS_TO_BENCH = new LinkedHashSet<AbstractMeter>();
67  
68      /** Result for all Benchmarks. */
69      private static BenchmarkResult benchRes;
70  
71      /** Boolean to be sure that the beforeFirstRun was not executed yet. */
72      private transient boolean beforeFirstRun;
73  
74      /** Boolean to be sure that the afterLastRun was not executed yet. */
75      private transient boolean afterLastRun;
76  
77      /**
78       * Corresponding BenchmarkElement of this executor to get the before/after
79       * methods.
80       */
81      private transient final BenchmarkMethod element;
82  
83      /**
84       * Private constructor, just setting the booleans and one element to get the
85       * before/after methods.
86       * 
87       * @param paramElement
88       *            BenchmarkElement to provide easy access to the before/after
89       *            methods.
90       */
91      private BenchmarkExecutor(final BenchmarkMethod paramElement) {
92          beforeFirstRun = false;
93          afterLastRun = false;
94          element = paramElement;
95      }
96  
97      /**
98       * Getting the executor corresponding to a BenchmarkElement.
99       * 
100      * @param meth
101      *            for the executor. If the underlaying {@link Method} was not
102      *            registered, a new mapping-entry will be created.
103      * @return the BenchmarkExecutor corresponding to the Method of the
104      *         BenchmarkElement
105      */
106     public static BenchmarkExecutor getExecutor(final BenchmarkElement meth) {
107         if (benchRes == null) {
108             throw new IllegalStateException("Call initialize method first!");
109         }
110 
111         // check if new instance needs to be created
112         if (!EXECUTOR.containsKey(meth.getMeth())) {
113             EXECUTOR.put(meth.getMeth(), new BenchmarkExecutor(meth.getMeth()));
114         }
115 
116         // returning the executor
117         return EXECUTOR.get(meth.getMeth());
118 
119     }

121     /**
122      * Initializing the executor.
123      * 
124      * @param meters
125      *            to be benched
126      * @param result
127      *            to be stored to
128      */
129     public static void initialize(final Set<AbstractMeter> meters, final BenchmarkResult result) {
130         METERS_TO_BENCH.clear();
131         METERS_TO_BENCH.addAll(meters);
132         EXECUTOR.clear();
133         benchRes = result;
134 
135     }

137     /**
138      * Executing the {@link BeforeFirstRun}-annotated methods (if still wasn't)
139      * and the {@link BeforeEachRun} methods.
140      * 
141      * @param obj
142      *            the object of the class where the bench runs currently in.
143      */
144     public void executeBeforeMethods(final Object obj) {
145 
146         // invoking once the beforeFirstRun-method
147         if (!beforeFirstRun) {
148             beforeFirstRun = true;
149 
150             Method beforeFirst = null;
151             try {
152                 beforeFirst = element.findBeforeFirstRun();
153             } catch (final PerfidixMethodCheckException e) {
154                 benchRes.addException(e);
155             }
156             if (beforeFirst != null) {
157                 checkAndExectuteBeforeAfters(obj, beforeFirst, BeforeFirstRun.class);
158             }
159         }
160 
161         // invoking the beforeEachRun-method
162         Method beforeEach = null;
163         try {
164             beforeEach = element.findBeforeEachRun();
165         } catch (final PerfidixMethodCheckException e) {
166             benchRes.addException(e);
167         }
168         if (beforeEach != null) {
169             checkAndExectuteBeforeAfters(obj, beforeEach, BeforeEachRun.class);
170         }
171 
172     }
173 
174     /**
175      * Execution of bench method. All data is stored corresponding to the
176      * meters.
177      * 
178      * @param objToExecute
179      *            the instance of the benchclass where the method should be
180      *            executed with.
181      */
182     public void executeBench(final Object objToExecute) {
183 
184         final double[] meterResults = new double[METERS_TO_BENCH.size()];
185 
186         final Method meth = element.getMethodToBench();
187 
188         int meterIndex1 = 0;
189         int meterIndex2 = 0;
190         for (final AbstractMeter meter : METERS_TO_BENCH) {
191             meterResults[meterIndex1] = meter.getValue();
192             meterIndex1++;
193         }
194 
195         final PerfidixMethodInvocationException res = invokeMethod(objToExecute, meth, Bench.class);
196 
197         for (final AbstractMeter meter : METERS_TO_BENCH) {
198             meterResults[meterIndex2] = meter.getValue() - meterResults[meterIndex2];
199             meterIndex2++;
200         }
201 
202         if (res == null) {
203             meterIndex1 = 0;
204             for (final AbstractMeter meter : METERS_TO_BENCH) {
205                 benchRes.addData(element.getMethodToBench(), meter, meterResults[meterIndex1]);
206                 meterIndex1++;
207             }
208         } else {
209             benchRes.addException(res);
210         }
211 
212     }

214     /**
215      * Executing the {@link AfterLastRun}-annotated methods (if still wasn't)
216      * and the {@link AfterEachRun} methods.
217      * 
218      * @param obj
219      *            the object of the class where the bench runs currently in.
220      */
221     public void executeAfterMethods(final Object obj) {
222 
223         // invoking once the beforeFirstRun-method
224         if (!afterLastRun) {
225             afterLastRun = true;
226             Method afterLast = null;
227             try {
228                 afterLast = element.findAfterLastRun();
229             } catch (final PerfidixMethodCheckException e) {
230                 benchRes.addException(e);
231             }
232             if (afterLast != null) {
233                 checkAndExectuteBeforeAfters(obj, afterLast, AfterLastRun.class);
234             }
235 
236         }
237 
238         // invoking the beforeEachRun-method
239         Method afterEach = null;
240         try {
241             afterEach = element.findAfterEachRun();
242         } catch (final PerfidixMethodCheckException e) {
243             benchRes.addException(e);
244         }
245         if (afterEach != null) {
246             checkAndExectuteBeforeAfters(obj, afterEach, AfterEachRun.class);
247         }
248 
249     }

251     /**
252      * Checking and executing several before/after methods.
253      * 
254      * @param obj
255      *            on which the execution should take place
256      * @param meth
257      *            to be executed
258      * @param anno
259      *            the related annotation
260      */
261     private void checkAndExectuteBeforeAfters(final Object obj, final Method meth,
262         final Class<? extends Annotation> anno) {
263         final PerfidixMethodCheckException checkExc = checkMethod(obj, meth, anno);
264         if (checkExc == null) {
265             final PerfidixMethodInvocationException invoExc = invokeMethod(obj, meth, anno);
266             if (invoExc != null) {
267                 benchRes.addException(invoExc);
268             }
269 
270         } else {
271             benchRes.addException(checkExc);
272         }
273     }

275     /**
276      * Method to invoke a reflective invokable method.
277      * 
278      * @param obj
279      *            on which the execution takes place
280      * @param meth
281      *            to be executed
282      * @param relatedAnno
283      *            related annotation for the execution
284      * @return {@link PerfidixMethodInvocationException} if invocation fails,
285      *         null otherwise.
286      */
287     public static PerfidixMethodInvocationException invokeMethod(final Object obj, final Method meth,
288         final Class<? extends Annotation> relatedAnno) {
289         final Object[] args = {};
290         PerfidixMethodInvocationException returnVal = null;
291         try {
292             meth.invoke(obj, args);
293         } catch (final IllegalArgumentException e) {
294             returnVal = new PerfidixMethodInvocationException(e, meth, relatedAnno);
295         } catch (final IllegalAccessException e) {
296             returnVal = new PerfidixMethodInvocationException(e, meth, relatedAnno);
297         } catch (final InvocationTargetException e) {
298             returnVal = new PerfidixMethodInvocationException(e.getCause(), meth, relatedAnno);
299         }
300         return returnVal;
301     }

303     /**
304      * Checking a method if it is reflective executable and if the mapping to
305      * the object fits.
306      * 
307      * @param obj
308      *            on which the execution takes place
309      * @param meth
310      *            to be checked
311      * @param anno
312      *            the related annotation for the check
313      * @return {@link PerfidixMethodCheckException} if something is wrong in the
314      *         mapping, null otherwise.
315      */
316     public static PerfidixMethodCheckException checkMethod(final Object obj, final Method meth,
317         final Class<? extends Annotation> anno) {
318         // check if the class of the object to be executed has the given method
319         boolean classMethodCorr = false;
320         for (final Method methodOfClass : obj.getClass().getDeclaredMethods()) {
321             if (methodOfClass.equals(meth)) {
322                 classMethodCorr = true;
323             }
324         }
325 
326         PerfidixMethodCheckException returnVal = null;
327         if (!classMethodCorr) {
328             returnVal =
329                 new PerfidixMethodCheckException(new IllegalStateException(new StringBuilder(
330                     "Object to execute ").append(obj).append(" is not having a Method named ").append(meth)
331                     .append(".").toString()), meth, anno);
332         }
333 
334         // check if the method is reflected executable
335         if (!BenchmarkMethod.isReflectedExecutable(meth)) {
336             returnVal =
337                 new PerfidixMethodCheckException(new IllegalAccessException(new StringBuilder(
338                     "Method to execute ").append(meth).append(" is not reflected executable.").toString()),
339                     meth, anno);
340         }
341         return returnVal;
342     }
