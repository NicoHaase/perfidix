27  package org.perfidix.socketadapter;
28  
29  import java.util.HashMap;
30  import java.util.Map;
31  import java.util.Set;
32  
33  import org.perfidix.element.BenchmarkMethod;
34  import org.perfidix.exceptions.AbstractPerfidixMethodException;
35  import org.perfidix.exceptions.SocketViewException;
36  import org.perfidix.meter.AbstractMeter;
37  
38  /**
39   * This class creates the connection to the eclipse view via
40   * {@link SocketViewStub}. It contains the methods which update the view to
41   * inform about the bench process progress.
42   * 
43   * @author Lewandowski Lukas, University of Konstanz
44   */
45  public final class SocketViewProgressUpdater implements IUpdater
46  {
47  
48      private final transient SocketViewStub viewStub;
49      private transient int meterHash = 0;
50      private transient boolean regMeterHash = false;
51  
52      /**
53       * The constructor initializes the host and port for creation a client
54       * socket.
55       * 
56       * @param host
57       *            Host is the host name of the view provider.
58       * @param port
59       *            Port represent the port number of the eclipse view.
60       * @throws SocketViewException
61       */
62      public SocketViewProgressUpdater(final String host, final int port)
63              throws SocketViewException
64      {
65          String strubParam = host;
66          if (host == null || host.equals(""))
67          {
68              strubParam = "localhost";
69          }
70          viewStub = new SocketViewStub(strubParam, port);
71  
72      }
73  
74      /**
75       * This method initializes the values of the eclipse view and resets the
76       * progress bar.
77       * 
78       * @param mapping
79       *            a mapping with all methods to benchmark and the related runs
80       * @throws SocketViewException
81       */
82      public boolean initProgressView(final Map<BenchmarkMethod, Integer> mapping)
83              throws SocketViewException
84      {
85          if (mapping != null)
86          {
87              final Set<BenchmarkMethod> methodSet = mapping.keySet();
88  
89              final Map<String, Integer> finalMap = new HashMap<String, Integer>();
90              for (BenchmarkMethod benchmarkMethod : methodSet)
91              {
92  
93                  finalMap.put(benchmarkMethod.getMethodWithClassName(),
94                          mapping.get(benchmarkMethod));
95              }
96  
97              viewStub.initTotalBenchProgress(finalMap);
98          }
99          return true;
100     }
101 
102     /**
103      * This method notifies the eclipse view which element is currently benched.
104      * 
105      * @param meter
106      *            The current meter.
107      * @param name
108      *            This param represents the java element which is currently
109      *            benched and which is fully qualified.
110      * @throws SocketViewException
111      */
112     public boolean updateCurrentElement(final AbstractMeter meter,
113             final String name) throws SocketViewException
114     {
115         if (meter != null && !regMeterHash)
116         {
117             registerFirstMeterHash(meter);
118         }
119         if (name != null && meter.hashCode() == (getRegMeter()))
120         {
121             viewStub.updateCurrentRun(name);
122         }
123         return true;
124     }
125 
126     /**
127      * This method informs the view that an error occurred while benching the
128      * current element.
129      * 
130      * @param name
131      *            Element represents the java element which has not been
132      *            executed successfully.
133      * @param exception
134      *            The exception caused by the element.
135      * @throws SocketViewException
136      */
137     public boolean updateErrorInElement(final String name,
138             final Exception exception) throws SocketViewException
139     {
140         if (name != null && exception != null)
141         {
142             if (exception instanceof AbstractPerfidixMethodException)
143             {
144                 final AbstractPerfidixMethodException exc = (AbstractPerfidixMethodException) exception;
145                 viewStub.updateError(name, exc.getExec().getClass()
146                         .getSimpleName());
147             }
148             if (exception instanceof SocketViewException)
149             {
150                 final SocketViewException viewException = (SocketViewException) exception;
151                 viewStub.updateError(name, viewException.getExc().getClass()
152                         .getSimpleName());
153             }
154         }
155         return true;
156     }
157 
158     /**
159      * This method notifies the view that all benches have been executed and the
160      * bench progress is finished.
161      * 
162      * @throws SocketViewException
163      */
164     public void finished() throws SocketViewException
165     {
166         viewStub.finishedBenchRuns();
167         regMeterHash = false;
168     }
169 
170     /**
171      * Register the meter hash for update the view, so only one time the current
172      * view element will be updated and not for every meter.
173      * 
174      * @param meterHash
175      *            The hash of the meter.
176      */
177     private void registerFirstMeterHash(final AbstractMeter meterHash)
178     {
179         this.meterHash = meterHash.hashCode();
180         regMeterHash = true;
181     }
182 
183     /**
184      * The meter hash.
185      * 
186      * @return The meter hash of the registered hash;
187      */
188     private int getRegMeter()
189     {
190         return meterHash;
191     }
