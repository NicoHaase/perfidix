42  /**
43   * The SocketAdapter is the main-class for registration of the classes that will
44   * be benched and creation of the socket stub to the ide view.
45   * 
46   * @author Lukas Lewandowski, University of Konstanz
47   * @author Sebastian Graf, University of Konstanz
48   */
49  public final class SocketAdapter {
50  
51      /** Instance for this run of the adapter */
52      private transient Benchmark benchmark;
53  
54      /** View instance for communicating with the perclipse plugin */
55      private transient final IUpdater view;
56  
57      /**
58       * public constructor.
59       * 
60       * @throws IllegalAccessException
61       * @throws InstantiationException
62       * @throws ClassNotFoundException
63       * @throws SocketViewException
64       */
65      public SocketAdapter(final IUpdater update, final String... classes) throws SocketViewException,
66          ClassNotFoundException, InstantiationException, IllegalAccessException {
67          view = update;
68  
69          // config adaptions for including the view
70          final AbstractConfig oldConf = Perfidix.getConfiguration(classes);
71          final AbstractOutput[] outputs = new AbstractOutput[oldConf.getListener().length + 1];
72          System.arraycopy(oldConf.getListener(), 0, outputs, 0, oldConf.getListener().length);
73          outputs[outputs.length - 1] = new SocketListener(view);
74  
75          // Building up the benchmark object
76          final AbstractConfig newConf =
77              new AbstractConfig(oldConf.getRuns(), oldConf.getMeters(), outputs, oldConf.getArrangement(),
78                  oldConf.getGcProb()) {
79              };
80          benchmark = new Benchmark(newConf);
81  
82      }
83  
84      /**
85       * Registering all classes and getting a mapping with the Methods and the
86       * corresponding overall runs
87       * 
88       * @param classNames
89       *            the names of the classes to be benched
90       */
91      public boolean registerClasses(final String... classNames) throws SocketViewException {
92          try {
93              benchmark = Perfidix.setUpBenchmark(classNames, benchmark);
94  
95              final Map<BenchmarkMethod, Integer> vals = benchmark.getNumberOfMethodsAndRuns();
96              return view.initProgressView(vals);
97          } catch (final ClassNotFoundException e2) {
98              return view.updateErrorInElement(e2.toString(), e2);
99          }
100     }
101 
102     /**
103      * This method starts the bench progress with the registered classes.
104      * 
105      * @throws SocketViewException
106      */
107     public boolean runBenchmark() throws SocketViewException {
108         final BenchmarkResult res = benchmark.run();
109         new TabularSummaryOutput().visitBenchmark(res);
110         view.finished();
111         return true;
112     }
113 
114     /**
115      * Main method for invoking benchs with classes as strings.
116      * 
117      * @param args
118      *            the classes
119      */
120     public static void main(final String[] args) {
121         // init of the connection to the plugin
122         int viewPort = 0;
123         final List<String> classList = new ArrayList<String>();
124         for (int i = 0; i < args.length; i++) {
125             if (args[i].equals("-Port")) {
126                 if (args[i + 1] != null) {
127                     viewPort = Integer.parseInt(args[i + 1]);
128                 }
129                 break;
130             } else {
131                 classList.add(args[i]);
132             }
133         }
134         try {
135 
136             final IUpdater updater = new SocketViewProgressUpdater(null, viewPort);
137 
138             final SocketAdapter adapter =
139                 new SocketAdapter(updater, classList.toArray(new String[classList.size()]));
140 
141             adapter.registerClasses(classList.toArray(new String[classList.size()]));
142             adapter.runBenchmark();
143         } catch (final SocketViewException e) {
144             throw new IllegalStateException(e);
145         } catch (ClassNotFoundException e) {
146             throw new IllegalStateException(e);
147         } catch (InstantiationException e) {
148             throw new IllegalStateException(e);
149         } catch (IllegalAccessException e) {
150             throw new IllegalStateException(e);
151         }
152     }
