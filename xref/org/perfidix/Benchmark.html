50  /**
51   * Class to hold all classes which want to be benchmarked.
52   * <p>
53   * 
54   * <pre>
55   * 
56   * 
57   * public class MyBenchmark {
58   * 
59   *     public static void main(final String[] args) {
60   *         // Build up own config
61   *         final AbstractConfig config = new MyConfig();
62   * 
63   *         // Initialise benchmark with config
64   *         final Benchmark benchmark = new Benchmark(config);
65   * 
66   *         // Adding Classes to benchmark
67   *         benchmark.add(Class1.class);
68   *         benchmark.add(Class2.class);
69   *         benchmark.add(Object3); // of Class3, every class is allowed to be inserted only once
70   * 
71   *         // Running benchmark
72   *         final BenchmarkResult result = benchmark.run();
73   * 
74   *         // Printing out Result in a Tabular, every AbstractOutput implementing class is allowed.
75   *         new TabularSummaryOutput().visitBenchmark(result);
76   * 
77   *     }
78   * }
79   * 
80   * </pre>
81   * 
82   * </p>
83   * 
84   * @see AbstractConfig
85   * @see BenchmarkResult
86   * @author Sebastian Graf, University of Konstanz
87   */

105     /**
106      * Constructor with a fixed set of used meters.
107      * 
108      * @param paramConf
109      *            Configuration for Benchmark
110      */
111     public Benchmark(final AbstractConfig paramConf) {
112         conf = paramConf;
113         this.meters = new LinkedHashSet<AbstractMeter>();
114         this.clazzes = new LinkedHashSet<Class<?>>();
115         this.objects = new LinkedHashSet<Object>();
116 
117         for (final AbstractMeter meter : conf.getMeters()) {
118             this.meters.add(meter);
119         }

129     /**
130      * Adding a class to bench to this benchmark. This class should contain
131      * benchmarkable methods, otherwise it will be ignored.
132      * 
133      * @param clazz
134      *            to be added.
135      */
136     public void add(final Class<?> clazz) {
137         if (this.clazzes.contains(clazz)) {
138             throw new IllegalArgumentException("Only one class-instance per benchmark allowed");
139         } else {
140             this.clazzes.add(clazz);
141         }
142     }

144     /**
145      * Adding a already instantiated objects to benchmark. Per benchmark, only
146      * one objects of each class is allowed.
147      * 
148      * @param obj
149      *            to be added
150      */
151     public void add(final Object obj) {
152         final Class<?> clazz = obj.getClass();
153 
154         if (this.clazzes.contains(clazz)) {
155             throw new IllegalArgumentException("Only one class-instance per benchmark allowed");
156         } else {
157             this.clazzes.add(clazz);
158             this.objects.add(obj);
159         }
160     }

162     /**
163      * Getting the number of all methods and all runs
164      * 
165      * @return a map with all methods and the runs.
166      */
167     public Map<BenchmarkMethod, Integer> getNumberOfMethodsAndRuns() {
168         final Map<BenchmarkMethod, Integer> returnVal = new HashMap<BenchmarkMethod, Integer>();
169         final Set<BenchmarkMethod> meths = getBenchmarkMethods();
170         for (final BenchmarkMethod meth : meths) {
171             int numberOfRuns = BenchmarkMethod.getNumberOfAnnotatedRuns(meth.getMethodToBench());
172             if (numberOfRuns == Bench.NONE_RUN) {
173                 numberOfRuns = conf.getRuns();
174             }
175             returnVal.put(meth, numberOfRuns);
176         }
177         return returnVal;
178     }

180     /**
181      * Running this benchmark
182      * 
183      * @return {@link BenchmarkResult} the result in an {@link BenchmarkResult} container.
184      */
185     public BenchmarkResult run() {
186         final BenchmarkResult res = new BenchmarkResult(conf.getListener());
187         BenchmarkExecutor.initialize(meters, res);
188 
189         // getting Benchmarkables
190         final Set<BenchmarkElement> elements = getBenchmarkElements();
191 
192         // arranging them
193         final AbstractMethodArrangement arrangement =
194             AbstractMethodArrangement.getMethodArrangement(elements, conf.getArrangement());
195 
196         // instantiate methods
197         final Map<Class<?>, Object> instantiatedObj = instantiateMethods(res);
198 
199         // getting the mapping and executing beforemethod
200         final Map<Class<?>, Object> objectsToExecute = executeBeforeBenchClass(instantiatedObj, res);
201 
202         // executing the bench for the arrangement
203         for (final BenchmarkElement elem : arrangement) {
204             // invoking gc if possible
205             if (RAN.nextDouble() < conf.getGcProb()) {
206                 System.gc();
207             }
208 
209             final BenchmarkExecutor exec = BenchmarkExecutor.getExecutor(elem);
210 
211             final Object obj = objectsToExecute.get(elem.getMeth().getMethodToBench().getDeclaringClass());
212             // check needed because of failed initialization of objects
213             if (obj != null) {
214                 exec.executeBeforeMethods(obj);
215                 exec.executeBench(obj);
216                 exec.executeAfterMethods(obj);
217             }
218         }
219 
220         // cleaning up methods to benchmark
221         tearDownObjectsToExecute(objectsToExecute, res);
222         return res;
223     }

225     /**
226      * Setting up executable objects for all registered classes and executing {@link BeforeBenchClass}
227      * annotated methods. If an {@link Exception} occurs, this failure will be stored in the
228      * {@link BenchmarkResult} and
229      * the class will not be instantiated
230      * 
231      * @param res
232      *            {@link BenchmarkResult} for storing possible failures.
233      * @return a mapping with class->objects for all registered classes-
234      */
235     private Map<Class<?>, Object> instantiateMethods(final BenchmarkResult res) {
236         // datastructure initialization for all objects
237         final Map<Class<?>, Object> objectsToUse = new Hashtable<Class<?>, Object>();
238 
239         // generating including already instaniated objects
240         for (final Object obj : this.objects) {
241             final Class<?> clazz = obj.getClass();
242             objectsToUse.put(clazz, obj);
243         }
244 
245         // generating objects for each registered class
246         for (final Class<?> clazz : clazzes) {
247             // generating a new instance on which the benchmark will be
248             // performed if there isn't a user generated one
249             if (!objectsToUse.containsKey(clazz)) {
250                 try {
251                     final Object obj = clazz.newInstance();
252                     objectsToUse.put(clazz, obj);
253                     // otherwise adding an exception to the result
254                 } catch (final InstantiationException e) {
255                     res.addException(new PerfidixMethodInvocationException(e, BeforeBenchClass.class));
256                 } catch (final IllegalAccessException e) {
257                     res.addException(new PerfidixMethodInvocationException(e, BeforeBenchClass.class));
258                 }
259 
260             }
261         }
262 
263         return objectsToUse;
264     }

266     /**
267      * Executing beforeBenchClass if present.
268      * 
269      * @param instantiatedObj
270      *            with the instantiatedObj;
271      * @param res
272      *            where the Exceptions should be stored to
273      * @return valid instances with valid beforeCall
274      */
275     private Map<Class<?>, Object> executeBeforeBenchClass(final Map<Class<?>, Object> instantiatedObj,
276         final BenchmarkResult res) {
277 
278         final Map<Class<?>, Object> returnVal = new Hashtable<Class<?>, Object>();
279 
280         // invoking before bench class
281         for (final Class<?> clazz : instantiatedObj.keySet()) {
282 
283             final Object objectToUse = instantiatedObj.get(clazz);
284             // ..the search for the beforeClassMeth begins...
285             Method beforeClassMeth = null;
286             boolean continueVal = true;
287             try {
288                 beforeClassMeth =
289                     BenchmarkMethod.findAndCheckAnyMethodByAnnotation(clazz, BeforeBenchClass.class);
290                 // ... and if this search is throwing an exception, the
291                 // exception will be added and a flag is set to break up
292             } catch (final PerfidixMethodCheckException e) {
293                 res.addException(e);
294                 continueVal = false;
295             }
296             // if everything worked well...
297             if (continueVal) {
298                 if (beforeClassMeth == null) {
299                     // ...either the objects is directly mapped to the class
300                     // for executing the benches
301                     returnVal.put(clazz, objectToUse);
302                 } else {
303                     // ... or the beforeMethod will be executed and a
304                     // possible exception stored to the result...
305                     final PerfidixMethodCheckException beforeByCheck =
306                         BenchmarkExecutor.checkMethod(objectToUse, beforeClassMeth, BeforeBenchClass.class);
307                     if (beforeByCheck == null) {
308                         final PerfidixMethodInvocationException beforeByInvok =
309                             BenchmarkExecutor.invokeMethod(objectToUse, beforeClassMeth,
310                                 BeforeBenchClass.class);
311                         if (beforeByInvok == null) {
312                             returnVal.put(clazz, objectToUse);
313                         } else {
314                             res.addException(beforeByInvok);
315                         }
316                     } else {
317                         res.addException(beforeByCheck);
318                     }
319                 }
320             }
321         }
322         return returnVal;
323     }

325     /**
326      * Tear down executable objects for all registered classes and executing {@link AfterBenchClass} annotated
327      * methods.
328      * 
329      * @param objects
330      *            a mapping with class->objects to be teared down
331      * @param res
332      *            the {@link BenchmarkResult} for storing possible failures.
333      */
334     private void tearDownObjectsToExecute(final Map<Class<?>, Object> objects, final BenchmarkResult res) {
335 
336         // executing tearDown for all clazzes registered in given Map
337         for (final Class<?> clazz : objects.keySet()) {
338             final Object objectToUse = objects.get(clazz);
339             if (objectToUse != null) {
340                 // executing AfterClass for all objects.
341                 Method afterClassMeth = null;
342                 try {
343                     afterClassMeth =
344                         BenchmarkMethod.findAndCheckAnyMethodByAnnotation(clazz, AfterBenchClass.class);
345                 } catch (final PerfidixMethodCheckException e) {
346                     res.addException(e);
347                 }
348                 // if afterClassMethod exists, the method will be executed and
349                 // possible failures will be stored in the BenchmarkResult
350                 if (afterClassMeth != null) {
351                     final PerfidixMethodCheckException afterByCheck =
352                         BenchmarkExecutor.checkMethod(objectToUse, afterClassMeth, AfterBenchClass.class);
353                     if (afterByCheck == null) {
354                         final PerfidixMethodInvocationException afterByInvok =
355                             BenchmarkExecutor
356                                 .invokeMethod(objectToUse, afterClassMeth, AfterBenchClass.class);
357                         if (afterByInvok != null) {
358                             res.addException(afterByInvok);
359                         }
360                     } else {
361                         res.addException(afterByCheck);
362                     }
363                 }
364             }
365         }
366     }

368     /**
369      * Getting all Benchmarkable methods out of the registered class.
370      * 
371      * @return a Set with {@link BenchmarkMethod}
372      */
373     public Set<BenchmarkMethod> getBenchmarkMethods() {
374         // Generating Set for returnVal
375         final Set<BenchmarkMethod> elems = new LinkedHashSet<BenchmarkMethod>();
376         // Getting all Methods and testing if its benchmarkable
377         for (final Class<?> clazz : clazzes) {
378             for (final Method meth : clazz.getDeclaredMethods()) {
379                 // Check if benchmarkable, if so, insert to returnVal;
380                 if (BenchmarkMethod.isBenchmarkable(meth)) {
381                     final BenchmarkMethod benchmarkMeth = new BenchmarkMethod(meth);
382                     elems.add(benchmarkMeth);
383                 }
384             }
385         }
386         return elems;
387     }

389     /**
390      * Getting all benchmarkable objects out of the registered classes with the
391      * annotated number of runs.
392      * 
393      * @return a Set with {@link BenchmarkMethod}
394      */
395     public Set<BenchmarkElement> getBenchmarkElements() {
396 
397         // Generating Set for returnVal
398         final Set<BenchmarkElement> elems = new LinkedHashSet<BenchmarkElement>();
399 
400         final Set<BenchmarkMethod> meths = getBenchmarkMethods();
401 
402         for (final BenchmarkMethod meth : meths) {
403             int numberOfRuns = BenchmarkMethod.getNumberOfAnnotatedRuns(meth.getMethodToBench());
404             if (numberOfRuns == Bench.NONE_RUN) {
405                 numberOfRuns = conf.getRuns();
406             }
407 
408             // getting the number of runs and adding this number of
409             // elements to the set to be evaluated.
410             for (int i = 0; i < numberOfRuns; i++) {
411                 elems.add(new BenchmarkElement(meth));
412             }
413         }
414 
415         return elems;
416     }
